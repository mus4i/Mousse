Properties后缀文件

'''
    Properties p = new Properties();
    p.load(new FileReader("Test_1028_mousse\\src\\UserTest.properties")); //读出目标数据
    System.out.println(p.getProperty("name"));  //根据键读取值
    System.out.println(p.getProperty("age"));
    System.out.println(p.getProperty("score"));
    Set<String> strs = p.stringPropertyNames();

    for (String str : strs) {
        System.out.println(str+"----"+p.getProperty(str));  //遍历键值对
    }

    p.forEach((k, v)-> {
        System.out.println(k +"--"+v);   //遍历键值对
    });

'''

dom4j框架的使用

    '''
    SAXReader saxReader = new SAXReader();
    Document r = saxReader.read("Test_1028_mousse\\hehee.xml");
    Element root = r.getRootElement();  //从文档解析xml的全部数据
    System.out.println(root.getName()); //读取xml的根标签
    List<Element> elements = root.elements();  //获取root下的全部一级标签
    //        List<Element> u = root.elements("user"); //获取指定的全部目录
    for (Element element : elements) {
        System.out.println(element.getName());

    }
    System.out.println(root.element("people").getText());  //获取目录下指定的一级标签内容
    System.out.println(root.element("user").elementText("age")); //这样写默认获取第一个user目录， elementText获取目录内容

    Element user = root.element("user");
    Attribute id = user.attribute("id");
    System.out.println(id.getName());   //获取id名字
    System.out.println(id.getValue());  //获取id的值
    System.out.println(user.attributeValue("id"));   //同上。 获取id的值

    List<Attribute> attributes = user.attributes();
    for (Attribute attribute : attributes) {
        System.out.println(attribute.getName()+ "----"+attribute.getValue());  //获取第一个标签全部内容
    }

    System.out.println(user.elementText("name"));   //获取个标签下的值
    System.out.println(user.elementText("age"));   
    System.out.println(user.elementText("score"));
    '''
