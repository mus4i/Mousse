File对象一些常用方法

'''
    File f1 = new File("C:\\Users\\24084\\Desktop\\学习笔记\\MySQL笔记.txt");  //获取文件对象
    File f2 = new File("C:/Users/24084/Desktop/学习笔记/MySQL笔记.txt");  //用正斜杠也可。 只需一个/
    File f3 = new File("C:" + File.separator + "Users"+File.separator+ "24084"+
            File.separator+"Desktop"+File.separator+"学习笔记"+File.separator+"MySQL笔记.txt");  //也可用file类自带的常量分割， 分隔符由系统决定
    System.out.println(f1.length());    //返回文件的字节个数
    System.out.println(f2.length());    //返回文件的字节个数
    System.out.println(f3.length());    //返回文件的字节个数

    File f4 = new File("C:/Users/24084/Desktop/学习笔记");  //也可获取文件夹对象
    System.out.println(f4.length());          //获取文件夹本身的大小， 并不包括文件夹里所有的内容

    System.out.println(f4.exists());  //判断文件或文件夹是否存在

    File f5 = new File("D:/呵呵哒");   //可以指代不存在的文件
    System.out.println(f5.exists());  //判断文件或文件夹是否存在

    File f6 = new File("D:\\LearrnMoreJavaTest\\Test_1028_mousse\\src\\com\\fileeTest\\呵呵哒.txt");//获取包里的文件， 这里是绝对路径

    File f7 = new File("Test_1028_mousse\\src\\com\\fileeTest\\呵呵哒.txt");//获取包里的文件， 这里是相对路径, 从包开始找， 也可从工程开始， 加上工程路径
    System.out.println(f7.length());

    System.out.println(f7.isFile());//判断该文件是否是文件

    System.out.println(f7.isDirectory()); //判断该文件是否是文件夹
    System.out.println(f4.isDirectory()); //判断该文件是否是文件夹

    System.out.println(f7.getName());  //获取文件的名字， 包括后缀

    System.out.println(f7.lastModified()); //获取文件的最后修改时间
    SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd hh:mm:ss");
    System.out.println(sdf.format(f7.lastModified()));  //解析后的时间

    System.out.println(f7.getPath());  //获取文件对象时， 使用的路径

    System.out.println(f7.getAbsolutePath());  //获取文件的绝对路径

'''

File类对象创建和删除文件的方法

'''
    File f1 = new File("D:\\LearrnMoreJavaTest\\Test_1028_mousse\\src\\com\\fileeTest\\呵呵哒2.txt");
    try {
        System.out.println(f1.createNewFile());   //创建文件， 成功返回true, 反之false
        System.out.println(f1.createNewFile());   //如果已经存在， 那么返回false， 不会再次创建
    } catch (IOException e) {
        e.printStackTrace();
    }

    File f2 = new File("D:\\LearrnMoreJavaTest\\Test_1028_mousse\\src\\com\\fileeTest\\呵呵哒文件夹");
    System.out.println(f2.mkdir());  //创建文件夹，成功返回true, 否则返回false 这个方法只能创建一级文件夹

    File f3 = new File("D:\\LearrnMoreJavaTest\\Test_1028_mousse\\src\\com\\fileeTest\\一级文件\\二级文件\\三级文件");
    System.out.println(f3.mkdirs());  //创建文件夹， 这个方法可以创建多级文件夹


    System.out.println(f2.delete());  //删除文件夹， 不能删除非空文件夹
    System.out.println(f3.delete());    //删除最外的三级文件

    File f4 = new File("D:\\LearrnMoreJavaTest\\Test_1028_mousse\\src\\com\\fileeTest");
    String[] list = f4.list();    //获取当前文件夹目录下所有一级文件名称， 并返回字符串数组
    System.out.println(Arrays.toString(list));

    File[] listFiles = f4.listFiles();   //获取当前文件夹目录下所有一级文件， 并返回文件数组
    for (File listFile : listFiles) {
        System.out.println(listFile.getAbsolutePath());

    }

'''

File类搜索文件的方法

'''
    public static void searchFile(File file, String FileName) {
    if (file == null || !file.exists() || file.isFile()) {
        return;
    }

    File[] files = file.listFiles();
    if (files != null && files.length > 0) {
        for (File f : files) {
            if (f.isFile()) {
                if (f.getName().contains(FileName)) {
                    System.out.println("找到了：" + f.getAbsolutePath());

            }

        }
    }
}

File类删除多级文件的方法

'''
    public static void deleteFile(File file){
    if (file==null||!file.exists()){
        return;
    }
    if (file.isFile()){
        file.delete();
    }
    File[] files = file.listFiles();
    if (files!=null) {
        for (File f : files) {
            if (f.isFile()) {
                f.delete();
            } else {
                deleteFile(f);
            }
        }
    }
    file.delete();
}
'''
