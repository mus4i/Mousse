FileOutputStream类、

'''
    OutputStream f1 = new FileOutputStream("Test_1028_mousse/呵呵哒.txt");  //这是覆盖管道， 会覆盖原来的数据
    f1.write(97);
    f1.write('h');
    byte[] b = "呵呵哒呵呵哒呵呵哒".getBytes();   //getBytes方法将字符串拆分返回字符数组形式
    f1.write(b);
    f1.close();

    OutputStream f2 = new FileOutputStream("Test_1028_mousse/呵呵哒呵呵哒.txt", true); //这是追加管道， 不会覆盖原来的数据
    byte[] b2 = "呵呵哒呵呵哒呵呵".getBytes();
    f2.write(b2);
    f2.write("\r\n".getBytes());  //\n换行不一定兼容全部平台， \r\n兼容全部平台
    f2.write(b2);
    f2.close();
}
'''

'''
//复制案例
    InputStream f1 = new FileInputStream("C:\\Users\\24084\\Desktop\\娱乐\\FuX.jpg");

    OutputStream f2 = new FileOutputStream("C:\\Users\\24084\\Desktop\\娱乐\\FuXing.jpg");

    int len;
    byte[] b = new byte[1024];
    while ((len = f1.read(b))!= -1){
        f2.write(b, 0, len);
    }  
    f2.close();                    //先关闭后创建的流
    f1.close();                    //再关闭前创建的流
    System.out.println("复制完成");

'''

下面是规范的写法：
'''
    InputStream f1 = null;
    OutputStream f2 = null;
    try {
        f1 = new FileInputStream("C:\\Users\\24084\\Desktop\\娱乐\\FuX.jpg");

        f2 = new FileOutputStream("C:\\Users\\24084\\Desktop\\娱乐\\FuXingCopy2.jpg");

        int len;
        byte[] b = new byte[1024];
        while ((len = f1.read(b)) != -1) {
            f2.write(b, 0, len);
        }
    } catch (IOException e) {
        e.printStackTrace();
    } finally {
        if (f2 != null) {
            try {
                f2.close();                    //先关闭后创建的流
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
        if (f1 != null) {
            try {
                f1.close();                    //再关闭前创建的流
                System.out.println("复制完成");
            } catch (IOException e) {
                e.printStackTrace();
            }
        }

    }
'''

    f1.write("呵呵哒");
    f1.write("呵呵哒");
    f1.flush();   //字符流使用完毕必须刷新或者关闭， 否则不会写出数据
    //刷新后的流还可以继续使用
    f1.write("\r\n");
    f1.write("、、、、呵呵哒");
    f1.flush();   //写完之后必须刷新， 或者关闭
    f1.close();   //关闭流包含刷新流， 关闭后不能再使用流了。
}
